        // Fetch production data from GitHub

        async function fetchProductionData() {

            debugLog(`Starting fetch from: ${DATA_URL}`);

            

            try {

                // Add cache-busting query param

                const cacheBuster = '?_=' + new Date().getTime();

                const fullUrl = DATA_URL + cacheBuster;

                debugLog(`Full URL with cache buster: ${fullUrl}`);

                

                const response = await fetch(fullUrl);

                debugLog(`Fetch response status: ${response.status}`);

                

                if (!response.ok) {

                    throw new Error(`HTTP error! Status: ${response.status}`);

                }

                

                const text = await response.text();

                debugLog(`Received data length: ${text.length} characters`);

                debugLog(`First 100 characters: ${text.substring(0, 100)}...`);

                

                let data;

                try {

                    data = JSON.parse(text);

                    debugLog("Successfully parsed JSON data");

                } catch (parseError) {

                    debugLog(`Error parsing JSON: ${parseError.message}`);

                    throw parseError;

                }

                

                // Check if the data has actually changed by comparing timestamps

                if (!lastUpdateTime || lastUpdateTime !== data.timestamp) {

                    debugLog(`Data timestamp changed: ${data.timestamp}`);

                    debugLog(`Data has variants: ${data.variants && data.variants.length > 0}`);

                    debugLog(`Number of variants: ${data.variants ? data.variants.length : 0}`);

                    if (data.variants && data.variants.length > 0) {

                        debugLog(`First variant: ${data.variants[0].variant}`);

                    }

                    

                    productionData = data;

                    lastUpdateTime = data.timestamp;

                    

                    // Update last update time display

                    const updateTime = new Date(data.timestamp);

                    document.getElementById('last-update').textContent = 

                        'Last updated: ' + updateTime.toLocaleTimeString();

                    

                    // Update supervisor dropdown

                    updateSupervisorDropdown();

                    

                    // Filter variants by selected supervisor

                    filterVariantsBySupervisor();

                    

                    // Update variant count

                    document.getElementById('total-variants').textContent = filteredVariants.length;

                    debugLog(`Filtered variants count: ${filteredVariants.length}`);

                    

                    // Reset index if out of bounds

                    if (currentVariantIndex >= filteredVariants.length) {

                        currentVariantIndex = 0;

                    }

                    

                    // Display current variant

                    if (filteredVariants.length > 0) {

                        displayVariant(currentVariantIndex);

                    } else {

                        debugLog("No variants to display after filtering");

                    }

                } else {

                    debugLog("Data timestamp unchanged, not updating");

                }

            } catch (error) {

                debugLog(`Error fetching data: ${error.message}`);

                console.error('Error fetching data:', error);

            }

        }

        

        // Update supervisor dropdown

        function updateSupervisorDropdown() {

            const selectElement = document.getElementById('supervisor-select');

            const currentValue = selectElement.value;

            

            // Clear existing options except the 'All' option

            while (selectElement.options.length > 1) {

                selectElement.remove(1);

            }

            

            // Add supervisors from data

            if (productionData.supervisors && productionData.supervisors.length > 0) {

                debugLog(`Adding ${productionData.supervisors.length} supervisors to dropdown`);

                productionData.supervisors.forEach(supervisor => {

                    const option = document.createElement('option');

                    option.value = supervisor.name;

                    option.textContent = supervisor.name;

                    selectElement.appendChild(option);

                    debugLog(`Added supervisor: ${supervisor.name}`);

                });

            }

            

            // Restore selected value if it still exists, otherwise select 'all'

            if (Array.from(selectElement.options).some(opt => opt.value === currentValue)) {

                selectElement.value = currentValue;

            } else {

                selectElement.value = 'all';

                selectedSupervisor = 'all';

            }

        }

        

        // Filter variants by selected supervisor

        function filterVariantsBySupervisor() {

            debugLog(`Filtering variants by supervisor: ${selectedSupervisor}`);

            

            if (selectedSupervisor === 'all') {

                // Show all variants

                filteredVariants = productionData.variants || [];

                debugLog(`Showing all variants: ${filteredVariants.length}`);

            } else {

                // Find the selected supervisor

                const supervisor = productionData.supervisors.find(s => s.name === selectedSupervisor);

                

                if (supervisor && supervisor.variants) {

                    // Get the variants for this supervisor

                    const supervisorVariantKeys = Object.keys(supervisor.variants);

                    debugLog(`Supervisor has ${supervisorVariantKeys.length} variants`);

                    

                    // Filter the full variants list to match the supervisor's variants

                    filteredVariants = productionData.variants.filter(variant => 

                        supervisorVariantKeys.includes(variant.variant));

                    debugLog(`Filtered to ${filteredVariants.length} variants for supervisor`);

                } else {

                    filteredVariants = [];

                    debugLog("No variants found for selected supervisor");

                }

            }

            

            // Update the variant count

            document.getElementById('total-variants').textContent = filteredVariants.length;

        }

        

        // Update date and time

        function updateDateTime() {

            const now = new Date();

            

            // Format date

            const options = { year: 'numeric', month: 'long', day: 'numeric' };

            document.getElementById('current-date').textContent = now.toLocaleDateString('en-US', options);

            

            // Format time

            document.getElementById('current-time').textContent = now.toLocaleTimeString('en-US', {

                hour12: false,

                hour: '2-digit',

                minute: '2-digit',

                second: '2-digit'

            });

        }

        

        // Display data for the specified variant

        function displayVariant(index) {

            if (!filteredVariants || filteredVariants.length === 0) {

                debugLog("No variants to display");

                return; // No data to display yet

            }

            

            const variant = filteredVariants[index];

            debugLog(`Displaying variant: ${variant.variant}`);

            

            // Update variant information

            document.getElementById('variant-name').textContent = `Product Variant: ${variant.variant}`;

            document.getElementById('cycle-time').textContent = `Target Cycle Time: ${variant.cycleTime} sec/unit`;

            document.getElementById('current-variant-index').textContent = (index + 1);

            

            // Update totals display

            updateTotalsDisplay(variant);

            

            // Create hourly production chart

            createHourlyChart(variant);

        }

        

        // Update the totals display

        function updateTotalsDisplay(variant) {

            const totalsDisplay = document.getElementById('totals-display');

            totalsDisplay.innerHTML = '';

            

            // Current total

            const totalActual = document.createElement('div');

            totalActual.className = 'total-item';

            totalActual.innerHTML = `

                <div class="total-label">Current Total</div>

                <div class="total-value total-actual">${variant.currentTotal || 0}</div>

            `;

            totalsDisplay.appendChild(totalActual);

            

            // Daily target

            const totalTarget = document.createElement('div');

            totalTarget.className = 'total-item';

            totalTarget.innerHTML = `

                <div class="total-label">Daily Target</div>

                <div class="total-value total-target">${variant.dailyTarget || 0}</div>

            `;

            totalsDisplay.appendChild(totalTarget);

            

            // Progress percentage

            const progress = variant.currentTotal && variant.dailyTarget 

                ? Math.floor((variant.currentTotal / variant.dailyTarget) * 100) 

                : 0;

            

            const totalProgress = document.createElement('div');

            totalProgress.className = 'total-item';

            totalProgress.innerHTML = `

                <div class="total-label">Progress</div>

                <div class="total-value total-progress">${progress}%</div>

            `;

            totalsDisplay.appendChild(totalProgress);

            

            // Boxes info

            if (variant.devicesPerBox) {

                const boxesCompleted = variant.currentBoxTotal || Math.floor(variant.currentTotal / variant.devicesPerBox);

                const totalBoxes = variant.dailyBoxTarget || Math.ceil(variant.dailyTarget / variant.devicesPerBox);

                

                const boxInfo = document.createElement('div');

                boxInfo.className = 'total-item';

                boxInfo.innerHTML = `

                    <div class="total-label">Boxes</div>

                    <div class="total-value">${boxesCompleted} / ${totalBoxes}</div>

                `;

                totalsDisplay.appendChild(boxInfo);

            }

        }

        

        // Create the hourly production chart

        function createHourlyChart(variant) {

            const chartContainer = document.getElementById('hourly-data');

            chartContainer.innerHTML = '';

            

            // Get current hour to determine which hours to show

            const now = new Date();

            const currentHour = now.getHours();

            debugLog(`Current hour: ${currentHour}`);

            

            // Show hours from 7AM to current hour (or 6PM if after hours)

            const startHour = 7; // 7AM

            const endHour = Math.min(currentHour + 1, 19); // Current hour or 7PM

            debugLog(`Showing hours from ${startHour} to ${endHour}`);

            

            // Calculate maximum value for chart scaling

            let maxTarget = 0;

            let maxProduction = 0;

            

            for (let i = startHour; i < endHour; i++) {

                if (variant.hourlyTargetArray && variant.hourlyTargetArray[i]) {

                    maxTarget = Math.max(maxTarget, variant.hourlyTargetArray[i]);

                }

                if (variant.hourlyProduction && variant.hourlyProduction[i]) {

                    maxProduction = Math.max(maxProduction, variant.hourlyProduction[i]);

                }

            }

            

            const maxValue = Math.max(maxTarget * 1.5, maxProduction * 1.2, 10);

            debugLog(`Chart max value: ${maxValue}`);

            

            // Create hour columns

            for (let i = startHour; i < endHour; i++) {

                const hourLabel = i < 10 ? `0${i}:00` : `${i}:00`;

                

                const hourDiv = document.createElement('div');

                hourDiv.className = 'hour-column';

                

                const barContainer = document.createElement('div');

                barContainer.className = 'bar-container';

                

                // Add target line (if target exists for this hour)

                if (variant.hourlyTargetArray && variant.hourlyTargetArray[i]) {

                    const targetLine = document.createElement('div');

                    targetLine.className = 'target-line';

                    targetLine.style.bottom = `${(variant.hourlyTargetArray[i] / maxValue) * 100}%`;

                    

                    const targetLabel = document.createElement('div');

                    targetLabel.className = 'target-label';

                    targetLabel.textContent = variant.hourlyTargetArray[i];

                    targetLine.appendChild(targetLabel);

                    

                    barContainer.appendChild(targetLine);

                }

                

                // Add the production bar (if production exists for this hour)

                const hourValue = variant.hourlyProduction[i] || 0;

                const hourTarget = variant.hourlyTargetArray ? (variant.hourlyTargetArray[i] || 0) : 0;

                

                const bar = document.createElement('div');

                bar.className = 'bar';

                

                if (hourValue >= hourTarget && hourTarget > 0) {

                    bar.classList.add('above-target');

                } else if (hourValue > 0) {

                    bar.classList.add('below-target');

                }

                

                // Animate bar height with a delay

                setTimeout(() => {

                    bar.style.height = `${(hourValue / maxValue) * 100}%`;

                }, 100 * (i - startHour));

                

                const barValue = document.createElement('div');

                barValue.className = 'bar-value';

                barValue.textContent = hourValue;

                bar.appendChild(barValue);

                

                barContainer.appendChild(bar);

                hourDiv.appendChild(barContainer);

                

                // Add hour label

                const hourLabelDiv = document.createElement('div');

                hourLabelDiv.className = 'hour-label';

                hourLabelDiv.textContent = hourLabel;

                hourDiv.appendChild(hourLabelDiv);

                

                chartContainer.appendChild(hourDiv);

            }

        }

        

        // Cycle to the next variant

        function cycleVariants() {

            if (!filteredVariants || filteredVariants.length === 0) {

                return; // No data to cycle through yet

            }

            

            updateCycleProgress(0);

            

            currentVariantIndex = (currentVariantIndex + 1) % filteredVariants.length;

            debugLog(`Cycling to variant ${currentVariantIndex + 1} of ${filteredVariants.length}`);

            displayVariant(currentVariantIndex);

            

            // Animate progress bar

            animateProgress();

        }

        

        // Animate the progress bar for timing

        function animateProgress() {

            let progress = 0;

            const interval = 100; // Update every 100ms

            const steps = CYCLE_INTERVAL / interval;

            

            const progressInterval = setInterval(() => {

                progress += (100 / steps);

                if (progress > 100) {

                    progress = 100;

                    clearInterval(progressInterval);

                }

                updateCycleProgress(progress);

            }, interval);

        }

        

        // Update the cycle progress bar

        function updateCycleProgress(percent) {

            document.getElementById('cycle-progress').style.width = `${percent}%`;

        }

        window.onload = initDashboard;
